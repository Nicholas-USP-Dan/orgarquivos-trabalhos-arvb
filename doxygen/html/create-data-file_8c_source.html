129int create_data_file_from_csv(const char *input_filename, const char *output_filename){
130 // Abrir o arquivo .csv no modo leitura
131 FILE *csv_data_fptr;
132 if((csv_data_fptr = fopen(input_filename, "r")) == NULL){
133 return -1;
134 }
135
136 // Abrir o arquivo de dados binario no modo escrita binaria
137 FILE *data_bfile_fptr;
138 if((data_bfile_fptr = fopen(output_filename, "wb")) == NULL){
139 fclose(csv_data_fptr);
140 return -1;
141 }
142
143 // Atribuição do status
144 set_campoc('0', data_bfile_fptr);
145
146 // Pular cabecalho
147 fseek(data_bfile_fptr, HEADER_END_OFFSET, SEEK_SET);
148
149 // Ler a primeira linha do arquivo .csv (colunas)
150 char columns[BUFFER_SIZE];
151 fgets(columns, BUFFER_SIZE, csv_data_fptr);
152 TRIM_NEWLINE(columns);
153
154 // Verifica se as colunas do .csv correspondem ao do projeto
155 if(strcmp(columns, COLUMN_NAMES) != 0){
156 fclose(csv_data_fptr);
157 fclose(data_bfile_fptr);
158 return -1; // Posso retornar 1 para indicar um erro lógico, não de saída
159 }
160
161 // Contador da quantidade de registros
162 int32_t reg_count = 0;
163
164 while(1){
165 // Ler linha no csv
166 char line_buff[BUFFER_SIZE];
167 fgets(line_buff, sizeof(char) * BUFFER_SIZE, csv_data_fptr);
168
169 if(feof(csv_data_fptr)){
170 break;
171 }
172
173 TRIM_NEWLINE(line_buff);
174
175 // Processa a linha do csv e coloca o jogador no arquivo binario
176 JOGADOR j = process_jogador(line_buff);
177 add_reg_bfile(j, data_bfile_fptr);
178 free_jogador(&j);
179
180 reg_count++;
181 }
182
183 // Pega o valor do proxByteOffset (campo do cabecalho do arquivo)
184 int64_t prox_byte_offset = ftell(data_bfile_fptr);
185
186 // Atribuicao do cabecalho
187 fseek(data_bfile_fptr, 0, SEEK_SET);
188 initialize_cabecalho('1', -1, prox_byte_offset, reg_count, 0, data_bfile_fptr);
189
190 fclose(csv_data_fptr);
191 fclose(data_bfile_fptr);
192
193 return 0;
194}
